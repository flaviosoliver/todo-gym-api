src/modules/shared/utils/build-params.utils.ts

buildParams
Default value : (params?: ParamsDto): any => { const opt: any = { $or: [] }; if (params && params.name) { opt.$or.push({ name: { $regex: params.name, $options: 'i' } }); } if (params && params.focusMuscle?.length > 0) { opt.$or.push({ focusMuscle: { $in: params.focusMuscle } }); } else { opt.$or.focusMuscle = []; } return opt; }

Default value : async ( passwordLogin: string, passwordUser: string ): Promise<boolean> => { return await bcrypt.compare(passwordLogin, passwordUser); }
